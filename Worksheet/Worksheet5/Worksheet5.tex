	Show that
	\begin{enumerate}
		\item If $\displaystyle f(x) = x^n$, then $f'(x) = n x^{n-1}$
		      \vspace{5cm}
		\item $ (f'(x) + g(x))' = f'(x) + g'(x)$
		      \vspace{5cm}
		\item $ (f'(x) - g(x))' = f'(x) - g'(x)$
		      \vspace{5cm}
		\item  $\displaystyle (f(x)g(x))'  = f'(x) g(x) + f(x) g'(x)$
		      \vspace{5cm}
	\end{enumerate}
\end{question}
